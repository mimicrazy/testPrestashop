  459. 459     public function formatQuantityDiscounts ($specificPrices, $price, $taxRate)
  460. 460     {
  461. 461         foreach ($specificPrices as $key => &$row) {
  462. 462             $row['quantity'] = &$row['from_quantity'];
  463. 463             // The price may be directly set
  464. 464             if ($row['price'] != 0) {
  465. 465                 $cur_price = (Product::$_taxCalculationMethod == PS_TAX_EXC ? $row['price'] : $row['price'] *
  466. 466                  (1 + $taxRate / 100));
  467. 467                 if ($row['reduction_type'] == 'amount')
  468. 468                     $cur_price = Product::$_taxCalculationMethod == PS_TAX_INC ? $cur_price -
  469. 469                      $row['reduction'] : $cur_price -

  474. 474             } else {
  475. 475                 global $cookie;
  476. 476                 $id_currency = (int) $cookie->id_currency;

  478. 478                     $reduction_amount = $row['reduction'];
  479. 479                     if (! $row['id_currency'])
  480. 480                         $reduction_amount = Tools::convertPrice(
  481. 481                         $reduction_amount, $id_currency);
  482. 482                     $row['real_value'] = Product::$_taxCalculationMethod ==
  483. 483                      PS_TAX_INC ? $reduction_amount : $reduction_amount /
  484. 484                      (1 + $taxRate / 100);
  485. 485                 } else {
  486. 486                     $row['real_value'] = $row['reduction'] * 100;
  487. 487                 }
  488. 488             }
  489. 489             $row['nextQuantity'] = (isset($specificPrices[$key + 1]) ? (int) ($specificPrices[$key +
  490. 490              1]['from_quantity']) : - 1);
