  293. 293  function tb_parseQuery ( query ) {
  294. 294   var Params = {};
  295. 295   if ( ! query ) {return Params;}// return empty object
  296. 296   var Pairs = query.split(/[;&]/);
  297. 297   for ( var i = 0; i < Pairs.length; i++ ) {
  298. 298   var KeyVal = Pairs[i].split('=');
  299. 299   if ( ! KeyVal || KeyVal.length != 2 ) {continue;}
  300. 300   var key = unescape( KeyVal[0] );
  301. 301   var val = unescape( KeyVal[1] );
  302. 302   val = val.replace(/\+/g, ' ');
  303. 303   Params[key] = val;
  304. 304   }
  305. 305   return Params;
  306. 306  }
